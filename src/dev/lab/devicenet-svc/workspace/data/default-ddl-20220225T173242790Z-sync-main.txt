CREATE TABLE Devices (id VARCHAR(32) NOT NULL ,type VARCHAR(16) NOT NULL ,name VARCHAR(32) NOT NULL ,groupId VARCHAR(32)  ,so VARCHAR(32)  ,imei VARCHAR(16)  ,imei2 VARCHAR(16)  ,mac VARCHAR(24)  ,macBluetooth VARCHAR(24)  ,ip VARCHAR(24)  ,brand VARCHAR(32)  ,model VARCHAR(32)  ,serialNumber VARCHAR(32)  ,created TIMESTAMP  ,CONSTRAINT Devices_PK PRIMARY KEY (id),CONSTRAINT Devices_UK UNIQUE (name));
CREATE TABLE Components (id VARCHAR(50) NOT NULL ,deviceId VARCHAR(32) NOT NULL ,name VARCHAR(16) NOT NULL ,type VARCHAR(16) NOT NULL ,brand VARCHAR(32)  ,model VARCHAR(32)  ,serialNumber VARCHAR(32)  ,created TIMESTAMP  ,CONSTRAINT Components_PK PRIMARY KEY (id),CONSTRAINT Components_UK UNIQUE (deviceId,name));
CREATE TABLE DeviceStatuses (id serial,deviceId VARCHAR(32) NOT NULL ,journeyId INTEGER  ,cpu DECIMAL(10,4)  ,cpuTemperature DECIMAL(10,4)  ,batery DECIMAL(10,4)  ,wifiSignal DECIMAL(10,4)  ,time TIMESTAMP NOT NULL ,registred TIMESTAMP  ,latitude DECIMAL(10,4)  ,longitude DECIMAL(10,4)  ,altitude DECIMAL(10,4)  ,CONSTRAINT DeviceStatuses_PK PRIMARY KEY (id));
CREATE TABLE Journeys (id serial,deviceId VARCHAR(32) NOT NULL ,startId INTEGER  ,endId INTEGER  ,created TIMESTAMP  ,CONSTRAINT Journeys_PK PRIMARY KEY (id));
CREATE TABLE Files (id VARCHAR(255) NOT NULL ,type VARCHAR(16) NOT NULL ,deviceId VARCHAR(32) NOT NULL ,componentId VARCHAR(50)  ,startDate TIMESTAMP NOT NULL ,endDate TIMESTAMP NOT NULL ,created TIMESTAMP  ,CONSTRAINT Files_PK PRIMARY KEY (id));
CREATE TABLE Users (username VARCHAR(32) NOT NULL ,firstname VARCHAR(80) NOT NULL ,lastname VARCHAR(80) NOT NULL ,email VARCHAR(512) NOT NULL ,created TIMESTAMP  ,CONSTRAINT Users_PK PRIMARY KEY (username),CONSTRAINT Users_UK UNIQUE (email));
CREATE TABLE Groups (id VARCHAR(32) NOT NULL ,name VARCHAR(32) NOT NULL ,created TIMESTAMP  ,CONSTRAINT Groups_PK PRIMARY KEY (id),CONSTRAINT Groups_UK UNIQUE (name));
CREATE TABLE GroupUsers (id VARCHAR(64) NOT NULL ,username VARCHAR(32) NOT NULL ,groupId VARCHAR(32) NOT NULL ,role VARCHAR(32) NOT NULL ,CONSTRAINT GroupUsers_PK PRIMARY KEY (id),CONSTRAINT GroupUsers_UK UNIQUE (groupId,username));
